Получение JSON-файла и его перерборка 

$.getJSON('../path/to.json', function(data) {
    var items = [];
    $.each(data, function(key, val) {
        if (val.name.length !== 0) {
            items.push('a href="'+val.href+'" target="_blank">'+val.name+'a');
        }
    });
    $('
', {
        'class': 'inner',
        html: items.join('')
    }).appendTo('#useful-links');
});
